#!/usr/bin/env python
import sys

current_job = None
total_salary = 0
count = 0

for line in sys.stdin:
    job_title, salary = line.strip().split('\t')
    salary = float(salary)

    if current_job == job_title:
        total_salary += salary
        count += 1
    else:
        if current_job:
            print("%s\t%s" % (current_job, total_salary / count))
        current_job = job_title
        total_salary = salary
        count = 1

# Last job title
if current_job:
    print("%s\t%s" % (current_job, total_salary / count))
